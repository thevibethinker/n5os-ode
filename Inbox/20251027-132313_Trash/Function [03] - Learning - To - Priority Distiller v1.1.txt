<functionArtifact>
  <artifactTitle>Learning-to-Priority Distiller — v1.1</artifactTitle>

  <persona>
    <companionFile>Companion File – Vrijen Persona Analysis.txt</companionFile>
  </persona>

  <goal>
    Convert any learning source (article, research paper, thread, etc.) into:
    • a comprehensive narrative summary, and
    • a topic‑organised digest that implicitly maps insights to the Intellectual Priority Ontology,
      enriching each point with sufficient detail to stand alone.
  </goal>

  <primaryKPI>Coverage‑weighted Relevance: (mapped_ideas / salient_ideas) × avg_relevance</primaryKPI>

  <corePrinciples>
    <principle>Ontology‑Aware but Topic‑Facing</principle>
    <principle>Detailed Yet Efficient Exposition</principle>
    <principle>Self‑Healing</principle>
    <principle>Contradiction Detection</principle>
    <principle>Integration Assistant</principle>
    <principle>Socratic Expansion</principle>
  </corePrinciples>

  <inputs>
    <input>learning_source (required) — URL, PDF, or raw text</input>
    <input>optional_focus — sections or questions to emphasise</input>
    <input>max_topics — default 6</input>
    <input>output_format_pref — default “topic‑memo + JSON”</input>
  </inputs>

  <outputs>
    <output>Narrative Summary — 1–3 paragraphs capturing the source’s argument arc</output>
    <output>Topic Memo — for each detected topic:  
            • Headline sentence  
            • 2–4 detailed bullets (≈40–70 tokens each) including priority hints in‑line (e.g., “[P7]”)</output>
    <output>JSON ⇒ {topic: {priority_ids:[], learnings:[{text,rationale,score}]}}</output>
    <output>Coverage metrics + Integration summary + Socratic Expansion</output>
  </outputs>

  <steps>

    <step id="S0">
      <title>Acquire & Clean</title>
      <description>
        • Fetch/ocr → clean via Text Cleanup & Formatting.
      </description>
    </step>

    <step id="S1">
      <title>Load Ontology & Persona</title>
      <description>
        • Parse ontology & persona for scoring and tone.
      </description>
    </step>

    <step id="S2">
      <title>Macro Summary</title>
      <description>
        • Generate a coherent 1–3 paragraph narrative capturing thesis, evidence, and conclusion.
      </description>
    </step>

    <step id="S3">
      <title>Extract Learnings</title>
      <description>
        • Identify salient learnings (≤40) with ≥40 tokens detail where needed.
      </description>
    </step>

    <step id="S4">
      <title>Topic Clustering</title>
      <description>
        • Cluster learnings into ≤ max_topics using semantic similarity.
      </description>
    </step>

    <step id="S5">
      <title>Priority Mapping</title>
      <description>
        • For each learning, compute relevance to priorities (keyword+regex+llm_hint).  
        • Store mapping in JSON; annotate bullet text with inline “[P#]” hints.
      </description>
    </step>

    <step id="S6">
      <title>Assemble Topic Memo</title>
      <description>
        • For each topic: headline + detailed bullets with rationales.
      </description>
    </step>

    <step id="S_INT">
      <title>Integration Assistant</title>
      <description>
        • Link themes with previous digests; output Integration Summary (≤120 tokens).
      </description>
    </step>

    <step id="S_SE">
      <title>Socratic Expansion</title>
      <description>
        • Craft 3–5 reflective questions referencing priority hints and topic names.
      </description>
    </step>

    <step id="S_QC">
      <title>Quality Checks</title>
      <description>
        • Contradiction detector; self‑heal; token bounds (memo ≤2000 tokens).
      </description>
    </step>

  </steps>

  <nuances enabled="true">
    companion_file_reference, extraction_salience, self_healing,
    contradiction_detector, canvas_output, web_search,
    integration_assistant, socratic_expansion
  </nuances>

  <finalGoal>
    Deliver a richly detailed, topic‑organised digest (plus JSON) and a coherent
    narrative summary, while still grounding every insight in the Intellectual
    Priority Ontology for strategic alignment.
  </finalGoal>
</functionArtifact>
