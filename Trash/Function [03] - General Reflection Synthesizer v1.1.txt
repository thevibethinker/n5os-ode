<functionArtifact>
  <artifactTitle>Stream-of-Consciousness Reflection Synthesizer — v1.1</artifactTitle>

  <!-- ✧ Goal -->
  <goal>
    Convert raw, free-form reflections into a structured artifact that maximally
    serves the user’s stated priorities (decision-making, product feedback,
    concept distillation, journaling, etc.).
    The function auto-classifies the reflection’s objective, selects an outcome-aligned
    persona, chooses the best-fit format (see enumerated list below), and
    delivers the result in Markdown by default—unless the user specifies otherwise.
  </goal>

  <!-- ✧ Core Principles -->
  <corePrinciples>
    <principle>Outcome-First: infer or confirm the end-goal before processing.</principle>
    <principle>Objective Classification: quickly label the reflection type
               to steer persona & format choices.</principle>
    <principle>Adaptive Persona: select a domain expert best suited to the classified objective.</principle>
    <principle>Format-Fit: map objective → optimal structure (enumerated below).</principle>
    <principle>Iterative Collaboration: present a draft; refine until the user approves.</principle>
    <principle>Low-Confidence Trigger: if any key inference is low (&lt;0.35),
               auto-invoke SocraticClarifier with one targeted question rather than halting.</principle>
  </corePrinciples>

  <!-- ✧ Enumerated Output Formats -->
  <outputFormatMenu>
    <format id="decisionJournal">Decision Journal (Context · Options · Chosen · Rationale · Next-Steps)</format>
    <format id="feedbackMatrix">Feedback Matrix (Area · Issue · Recommendation · Impact)</format>
    <format id="insightBullets">Insight Bullet List (Key Ideas · Implications)</format>
    <format id="quoteBank">Quote Bank (Verbatim lines + timestamps)</format>
    <format id="sourceTracker">Source Tracker (References · Links · Mentions)</format>
    <format id="conceptMap">Concept Map Outline (Themes → Sub-themes → Connections)</format>
  </outputFormatMenu>

  <!-- ✧ Steps -->
  <steps>

    <step id="1">
      <title>Mandatory Interrogatory — Priorities & Context</title>
      <description>
        • Ask for current priorities / what matters most.<br/>
        • Auto-infer when unclear (AutoInferenceHelper).<br/>
        • Confirm default output (Markdown).<br/>
        • Capture hard constraints (e.g., include sources, omit quotes).
      </description>
    </step>

    <step id="1A">
      <title>Objective Classification (Auto)</title>
      <description>
        • Classify the reflection into one of: Decision, Product Feedback, Concept Exploration,
          Personal Journal, or Other.<br/>
        • Store the label for downstream persona & format mapping.
      </description>
    </step>

    <step id="2">
      <title>Persona & Format Selection</title>
      <description>
        • Choose an outcome-oriented persona matching the classified objective.<br/>
        • Select the optimal format from &lt;outputFormatMenu&gt;.<br/>
        • If confidence in either choice &lt; 0.35, trigger SocraticClarifier
          with a single clarifying question.
      </description>
    </step>

    <step id="3">
      <title>Reflection Transformation</title>
      <description>
        • Parse the transcript (chunk & summarize if &gt; 8 000 tokens).<br/>
        • Apply the persona’s analytical lens.<br/>
        • Render the deliverable in the chosen structure.
      </description>
    </step>

    <step id="4">
      <title>Present Draft & Solicit Feedback</title>
      <description>
        • Display the formatted output plus a note on persona & format chosen.<br/>
        • Invite refinements or alternative formats.
      </description>
    </step>

    <step id="5">
      <title>Finalize & Deliver</title>
      <description>
        • Apply edits, confirm satisfaction, output final artifact.<br/>
        • Log persona/format mapping for future auto-selection learning.
      </description>
    </step>

    <step id="6">
      <title>Soft Fail-Safe</title>
      <description>
        • Loop back to Step 1 if priorities remain ambiguous or contradictory.
      </description>
    </step>

  </steps>

  <!-- ✧ Outputs -->
  <outputs>
    <output>Markdown (default) or user-specified format containing the structured reflection.</output>
    <output>Lightweight learning log of persona/format choices (optional).</output>
  </outputs>

  <!-- ✧ Final Goal -->
  <finalGoal>
    Deliver an adaptive, persona-aligned, outcome-focused reflection artifact that
    the user can immediately act on or archive, with minimal follow-up tweaks.
  </finalGoal>

</functionArtifact>
