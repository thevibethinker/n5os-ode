```python
#!/usr/bin/env python3
"""Fail-safe context validators for close skills."""

import os
import sys
import yaml
from pathlib import Path

def load_session_state(convo_id: str) -> dict:
    """Load SESSION_STATE.md frontmatter for a conversation."""
    state_path = Path(f"/home/.z/workspaces/{convo_id}/SESSION_STATE.md")
    if not state_path.exists():
        return {}
    
    content = state_path.read_text()
    if not content.startswith('---'):
        return {}
    
    # Extract YAML frontmatter
    parts = content.split('---', 2)
    if len(parts) < 3:
        return {}
    
    try:
        return yaml.safe_load(parts[1]) or {}
    except:
        return {}

def detect_context(state: dict) -> str:
    """Detect conversation context type.
    
    Returns: "drop" | "build" | "thread"
    """
    # Pulse Drop (headless worker)
    if state.get('drop_id') or state.get('worker_id'):
        return "drop"
    
    # Build context (orchestrator or post-build)
    if state.get('build_slug') and not state.get('drop_id'):
        return "build"
    
    # Normal interactive thread
    return "thread"

def warn_wrong_skill(called: str, suggested: str, reason: str) -> None:
    """Print warning about wrong skill being called."""
    print(f"""
⚠️  WRONG SKILL DETECTED

You called: {called}
Suggested:  {suggested}
Reason:     {reason}

Run the suggested skill instead, or use --force to override.
""", file=sys.stderr)

def validate_thread_context(state: dict) -> tuple[bool, str]:
    """Validate context is appropriate for thread-close."""
    ctx = detect_context(state)
    
    if ctx == "drop":
        return False, "SESSION_STATE has drop_id — use drop-close instead"
    
    if ctx == "build":
        return False, "Build context detected — use build-close instead"
    
    return True, "OK"

def validate_drop_context(state: dict) -> tuple[bool, str]:
    """Validate context is appropriate for drop-close."""
    ctx = detect_context(state)
    
    if ctx == "thread":
        return False, "No drop context — use thread-close instead"
    
    if ctx == "build" and not state.get('drop_id'):
        return False, "Build orchestrator context — use build-close instead"
    
    if not state.get('build_slug'):
        return False, "Missing build_slug in SESSION_STATE"
    
    return True, "OK"

def validate_build_context(slug: str) -> tuple[bool, str]:
    """Validate context is appropriate for build-close."""
    build_dir = Path(f"/home/workspace/N5/builds/{slug}")
    
    if not build_dir.exists():
        return False, f"Build not found: {slug}"
    
    meta_path = build_dir / "meta.json"
    if not meta_path.exists():
        return False, f"No meta.json in build: {slug}"
    
    import json
    meta = json.loads(meta_path.read_text())
    
    # Check build is in terminal state
    if meta.get('status') not in ('complete', 'partial', 'failed'):
        return False, f"Build not finished (status: {meta.get('status')})"
    
    # Check there are deposits to synthesize
    deposits_dir = build_dir / "deposits"
    if not deposits_dir.exists() or not list(deposits_dir.glob("D*.json")):
        return False, "No deposits found to synthesize"
    
    return True, "OK"
```

```python
#!/usr/bin/env python3
"""Core close logic — main entry points for all close types."""

import json
import os
from datetime import datetime
from pathlib import Path
from typing import Optional

from . import guards, emoji, positions, content_library, aar, pii

def detect_tier(convo_id: str) -> int:
    """Auto-detect appropriate tier for thread close.
    
    Tier 1: Simple discussions (<3 artifacts)
    Tier 2: Standard work (3-10 artifacts, research)
    Tier 3: Builds, orchestrators, complex debug
    """
    workspace = Path(f"/home/.z/workspaces/{convo_id}")
    state = guards.load_session_state(convo_id)
    
    # Tier 3 signals
    if state.get('type') == 'build':
        return 3
    if state.get('mode') == 'orchestrator':
        return 3
    
    # Count artifacts
    artifact_count = 0
    for pattern in ['*.py', '*.md', '*.json', '*.ts']:
        artifact_count += len(list(workspace.glob(pattern)))
    
    if artifact_count >= 10:
        return 3
    elif artifact_count >= 3:
        return 2
    else:
        return 1

def run_thread_close(convo_id: str, tier: Optional[int] = None, dry_run: bool = False) -> int:
    """Run full thread close for interactive conversations.
    
    Returns: 0 on success, 1 on error
    """
    state = guards.load_session_state(convo_id)
    
    # Auto-detect tier if not specified
    if tier is None:
        tier = detect_tier(convo_id)
    
    print(f"Running thread-close (Tier {tier}) for {convo_id}")
    if dry_run:
        print("[DRY RUN] Would execute close steps")
        return 0
    
    results = {
        'tier': tier,
        'convo_id': convo_id,
        'timestamp': datetime.utcnow().isoformat(),
        'steps': []
    }
    
    # Tier 1+: Basic close
    results['steps'].append('session_state_audit')
    results['title'] = emoji.generate_title(state, convo_id)
    results['summary'] = "TODO: LLM generates summary"
    
    # Tier 2+: Decisions, next steps
    if tier >= 2:
        results['steps'].append('decision_extraction')
        results['steps'].append('next_steps')
        results['decisions'] = []  # TODO: LLM extracts
        results['next_steps'] = []  # TODO: LLM extracts
    
    # Tier 3: AAR, positions, content library
    if tier >= 3:
        results['steps'].append('aar_generation')
        results['steps'].append('position_extraction')
        results['steps'].append('content_library_scan')
        results['aar'] = aar.generate_thread_aar(convo_id, state)
        results['positions'] = positions.extract_from_conversation(convo_id)
        results['content_candidates'] = content_library.scan_conversation(convo_id)
    
    # PII audit (all tiers)
    results['steps'].append('pii_audit')
    pii.audit_conversation(convo_id)
    
    print(f"✓ Thread close complete (Tier {tier})")
    return 0

def run_drop_close(convo_id: str, drop_id: str, build_slug: str) -> int:
    """Run drop close for Pulse workers.
    
    Writes structured deposit, does NOT commit.
    """
    print(f"Running drop-close for {drop_id} in build {build_slug}")
    
    state = guards.load_session_state(convo_id)
    
    # Gather deposit data
    deposit = {
        'drop_id': drop_id,
        'convo_id': convo_id,
        'status': 'complete',  # TODO: Detect from conversation
        'completed_at': datetime.utcnow().isoformat(),
        'summary': "TODO: Extract from conversation",
        'artifacts': [],  # TODO: Scan workspace
        'learnings': "",  # TODO: Extract
        'concerns': "",  # TODO: Extract
        'decisions': []  # TODO: Extract key decisions
    }
    
    # Write deposit
    deposit_path = Path(f"/home/workspace/N5/builds/{build_slug}/deposits/{drop_id}.json")
    deposit_path.parent.mkdir(parents=True, exist_ok=True)
    deposit_path.write_text(json.dumps(deposit, indent=2))
    
    print(f"✓ Deposit written: {deposit_path}")
    return 0

def run_build_close(slug: str, dry_run: bool = False) -> int:
    """Run build-level close — synthesizes across all deposits.
    
    This replaces N per-worker closes with 1 build-level close.
    """
    print(f"Running build-close for {slug}")
    
    build_dir = Path(f"/home/workspace/N5/builds/{slug}")
    deposits_dir = build_dir / "deposits"
    
    # Load all deposits
    deposits = []
    for deposit_file in sorted(deposits_dir.glob("D*.json")):
        if '_filter' in deposit_file.name or '_forensics' in deposit_file.name:
            continue
        deposits.append(json.loads(deposit_file.read_text()))
    
    print(f"Found {len(deposits)} deposits to synthesize")
    
    if dry_run:
        print("[DRY RUN] Would synthesize deposits")
        return 0
    
    # Aggregate decisions
    all_decisions = []
    for d in deposits:
        all_decisions.extend(d.get('decisions', []))
    
    # Aggregate learnings
    all_learnings = [d.get('learnings', '') for d in deposits if d.get('learnings')]
    
    # Aggregate concerns
    all_concerns = [d.get('concerns', '') for d in deposits if d.get('concerns')]
    
    # Extract positions from aggregated context
    position_candidates = positions.extract_from_build(slug, deposits)
    
    # Scan for content library candidates
    content_candidates = content_library.scan_build(slug)
    
    # Generate build AAR
    build_aar = aar.generate_build_aar(slug, deposits, all_decisions)
    
    # Write outputs
    close_path = build_dir / "BUILD_CLOSE.md"
    close_content = f"""# Build Close: {slug}

## Summary

Synthesized {len(deposits)} deposits.

## Decisions ({len(all_decisions)})

{chr(10).join(f"- {d.get('decision', 'Unknown')}: {d.get('rationale', '')}" for d in all_decisions)}

## Learnings

{chr(10).join(f"- {l}" for l in all_learnings if l)}

## Concerns

{chr(10).join(f"- {c}" for c in all_concerns if c)}

## Position Candidates ({len(position_candidates)})

{chr(10).join(f"- {p}" for p in position_candidates)}

## Content Library Candidates ({len(content_candidates)})

{chr(10).join(f"- {c}" for c in content_candidates)}
"""
    close_path.write_text(close_content)
    
    # Write AAR
    aar_path = build_dir / "BUILD_AAR.md"
    aar_path.write_text(build_aar)
    
    print(f"✓ Build close complete")
    print(f"  - {close_path}")
    print(f"  - {aar_path}")
    return 0
```

```python
    'orchestrator': '🐙',
    'worker': '👷🏽‍♂️',
    'linked': '🔗',
}

# Content emojis (slot 3)
CONTENT_EMOJIS = {
    'build': '🏗️',
    'research': '🔎',
    'repair': '🛠️',
    'site': '🕸️',
    'log': '🪵',
    'content': '✍️',
    'reflection': '🪞',
    'social': '🤳',
    'data': '📊',
    'comms': '💬',
    'organize': '🗂️',
    'planning': '📝',
}

def detect_state(state: dict, convo_id: str) -> str:
    """Detect thread completion state."""
    status = state.get('status', 'unknown')
    if status == 'complete':
        return 'complete'
    if status in ('blocked', 'failed'):
        return 'failed'
    if status == 'paused':
        return 'paused'
    return 'complete'  # Default assumption at close time

def detect_type(state: dict) -> str:
    """Detect thread type."""
    if state.get('drop_id') or state.get('worker_id'):
        return 'worker'
    if state.get('mode') == 'orchestrator':
        return 'orchestrator'
    if state.get('parent_convo_id'):
        return 'linked'
    return 'normal'

def detect_content(state: dict, convo_id: str) -> str:
    """Detect primary content type from conversation."""
    conv_type = state.get('type', '')
    
    if conv_type == 'build':
        return 'build'
    if conv_type == 'research':
        return 'research'
    if conv_type == 'planning':
        return 'planning'
    
    # Could add more heuristics based on artifacts
    return 'build'  # Default

def generate_title(state: dict, convo_id: str, semantic_title: Optional[str] = None) -> str:
    """Generate full title with 3-slot emoji prefix.
    
    Format: MMM DD | {state} {type} {content} Semantic Title
    """
    from datetime import datetime
    
    # Date prefix
    date_str = datetime.now().strftime("%b %d")
    
    # 3 emoji slots
    state_emoji = STATE_EMOJIS.get(detect_state(state, convo_id), '✅')
    type_emoji = TYPE_EMOJIS.get(detect_type(state), '📌')
    content_emoji = CONTENT_EMOJIS.get(detect_content(state, convo_id), '🏗️')
    
    # Semantic title (would be LLM-generated)
    title = semantic_title or state.get('focus', 'Untitled')
    
    return f"{date_str} | {state_emoji} {type_emoji} {content_emoji} {title}"
```

```python
    """Scan conversation workspace for content library candidates."""
    workspace = Path(f"/home/.z/workspaces/{convo_id}")
    candidates = []

    for f in workspace.glob("**/*framework*.md"):

    candidates = []
    
    if build_dir.exists():

    return candidates
```

```python
#!/usr/bin/env python3
"""PII audit for conversations."""

import subprocess
from pathlib import Path

def audit_conversation(convo_id: str, auto_mark: bool = True) -> dict:
    """Run PII audit on conversation workspace.
    
    Wraps existing N5/scripts/conversation_pii_audit.py
    """
    script = Path("/home/workspace/N5/scripts/conversation_pii_audit.py")
    
    if not script.exists():
        return {'status': 'skipped', 'reason': 'PII audit script not found'}
    
    cmd = ['python3', str(script), '--convo-id', convo_id]
    if auto_mark:
        cmd.append('--auto-mark')
    
    result = subprocess.run(cmd, capture_output=True, text=True)
    
    return {
        'status': 'complete' if result.returncode == 0 else 'error',
        'stdout': result.stdout,
        'stderr': result.stderr
    }
```
