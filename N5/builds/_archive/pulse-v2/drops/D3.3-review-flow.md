```python
#!/usr/bin/env python3
"""
HITL review manager for Pulse v2.
Handles plan review flow: upload → notify → wait → process response.
"""

import json
import argparse
import subprocess
from pathlib import Path
from datetime import datetime

def run_cmd(cmd: list) -> dict:
    result = subprocess.run(cmd, capture_output=True, text=True)
    try:
        return json.loads(result.stdout)
    except:
        return {"error": result.stderr or result.stdout, "stdout": result.stdout}

def initiate_review(slug: str) -> dict:
    """
    Initiate plan review:
    1. Sync to Google Drive
    2. Get shareable link
    3. Send SMS notification
    4. Update status to plan_review
    """
    build_dir = Path(f"N5/builds/{slug}")
    meta_path = build_dir / "meta.json"
    
    if not meta_path.exists():
        return {"success": False, "error": "Build not found"}
    
    meta = json.loads(meta_path.read_text())
    
    # 1. Sync to Drive
    sync_result = run_cmd(["python3", "N5/pulse/outpost_manager.py", "sync", slug])
    
    if not sync_result.get("success"):
        # Graceful degradation: continue with local plan
        link = f"(Drive sync failed - review locally at N5/builds/{slug}/PLAN.md)"
    else:
        link = sync_result.get("shareable_link", "")
    
    # 2. Generate SMS summary
    plan_path = build_dir / "PLAN.md"
    plan_content = plan_path.read_text() if plan_path.exists() else ""
    
    # Extract key info for SMS
    title = meta.get("title", slug)
    task_type = meta.get("build_type", "unknown")
    drops_count = len(meta.get("drops", {}))
    
    sms_message = f"""📋 Plan ready: "{title}"
Type: {task_type}
Drops: {drops_count}

{link}

Reply:
• "go" or "approve" to build
• "revise: <feedback>" to adjust"""
    
    # 3. Update meta
    meta["status"] = "plan_review"
    meta["review"] = {
        "initiated_at": datetime.utcnow().isoformat() + "Z",
        "link": link,
        "response": None,
        "responded_at": None
    }
    meta_path.write_text(json.dumps(meta, indent=2))
    
    # 4. Update task queue
    task_id = meta.get("task_id")
    if task_id:
        run_cmd(["python3", "N5/pulse/queue_manager.py", "advance", task_id, "plan_review"])
    
    return {
        "success": True,
        "status": "awaiting_review",
        "sms_message": sms_message,
        "link": link,
        "note": "Send SMS to V with the message above"
    }

def process_review_response(slug: str, response: str) -> dict:
    """
    Process V's response to plan review.
    
    Responses:
    - "go", "approve", "yes", "👍" → Advance to building
    - "revise: <feedback>" → Feed back to plan generator
    - "cancel" → Mark task as cancelled
    """
    build_dir = Path(f"N5/builds/{slug}")
    meta_path = build_dir / "meta.json"
    
    if not meta_path.exists():
        return {"success": False, "error": "Build not found"}
    
    meta = json.loads(meta_path.read_text())
    response_lower = response.strip().lower()
    
    # Record response
    meta["review"]["response"] = response
    meta["review"]["responded_at"] = datetime.utcnow().isoformat() + "Z"
    
    # Determine action
    approval_signals = ["go", "approve", "approved", "yes", "👍", "lgtm", "ship it", "looks good"]
    
    if any(signal in response_lower for signal in approval_signals):
        # APPROVED → advance to building
        meta["status"] = "building"
        meta_path.write_text(json.dumps(meta, indent=2))
        
        # Update task queue
        task_id = meta.get("task_id")
        if task_id:
            run_cmd(["python3", "N5/pulse/queue_manager.py", "advance", task_id, "building"])
        
        return {
            "success": True,
            "action": "approved",
            "next_status": "building",
            "message": "Plan approved. Ready to start build."
        }
    
    elif response_lower.startswith("revise:"):
        # REVISION REQUESTED
        feedback = response[7:].strip()
        meta["review"]["revision_feedback"] = feedback
        meta["status"] = "planning"  # Back to planning
        meta_path.write_text(json.dumps(meta, indent=2))
        
        return {
            "success": True,
            "action": "revision_requested",
            "feedback": feedback,
            "next_status": "planning",
            "message": f"Revision requested: {feedback}. Regenerate plan and re-review."
        }
    
    elif response_lower == "cancel":
        meta["status"] = "cancelled"
        meta_path.write_text(json.dumps(meta, indent=2))
        
        task_id = meta.get("task_id")
        if task_id:
            run_cmd(["python3", "N5/pulse/queue_manager.py", "advance", task_id, "failed"])
        
        return {
            "success": True,
            "action": "cancelled",
            "message": "Build cancelled."
        }
    
    else:
        # Unknown response
        return {
            "success": False,
            "action": "unknown",
            "message": f"Unclear response: '{response}'. Reply 'go' to approve or 'revise: <feedback>' to adjust."
        }

def check_review_status(slug: str) -> dict:
    """Check current review status."""
    meta_path = Path(f"N5/builds/{slug}/meta.json")
    
    if not meta_path.exists():
        return {"error": "Build not found"}
    
    meta = json.loads(meta_path.read_text())
    review = meta.get("review", {})
    
    return {
        "status": meta.get("status"),
        "initiated_at": review.get("initiated_at"),
        "responded_at": review.get("responded_at"),
        "response": review.get("response"),
        "awaiting_response": review.get("responded_at") is None and meta.get("status") == "plan_review"
    }

def main():
    parser = argparse.ArgumentParser(description="Review Manager")
    subparsers = parser.add_subparsers(dest="command")
    
    # initiate
    init_parser = subparsers.add_parser("initiate", help="Start plan review")
    init_parser.add_argument("slug", help="Build slug")
    
    # respond
    resp_parser = subparsers.add_parser("respond", help="Process review response")
    resp_parser.add_argument("slug", help="Build slug")
    resp_parser.add_argument("response", help="V's response")
    
    # status
    status_parser = subparsers.add_parser("status", help="Check review status")
    status_parser.add_argument("slug", help="Build slug")
    
    args = parser.parse_args()
    
    if args.command == "initiate":
        result = initiate_review(args.slug)
    elif args.command == "respond":
        result = process_review_response(args.slug, args.response)
    elif args.command == "status":
        result = check_review_status(args.slug)
    else:
        parser.print_help()
        return
    
    print(json.dumps(result, indent=2))

if __name__ == "__main__":
    main()
```

def cmd_plan(args):
    """Generate plan from seeded task."""
    result = subprocess.run(
        ["python3", "N5/pulse/plan_generator.py", args.task_id],
        capture_output=True, text=True
    )
    data = json.loads(result.stdout)
    
    if data.get("success"):
        print(f"✅ Plan generated: {data['plan_path']}")
        print(f"   Drops: {data['drops_count']}")
        print(f"   Streams: {data['streams']}")
        print(f"\nNext: pulse.py review {args.task_id}")
    else:
        print(f"❌ Plan generation failed: {data.get('error')}")

def cmd_review(args):
    """Initiate HITL plan review."""
    result = subprocess.run(
        ["python3", "N5/pulse/review_manager.py", "initiate", args.slug],
        capture_output=True, text=True
    )
    data = json.loads(result.stdout)
    
    if data.get("success"):
        print("📋 Plan review initiated")
        print(f"\n--- SMS MESSAGE ---\n{data['sms_message']}\n---")
        print("\nWaiting for V's response...")
    else:
        print(f"❌ Review initiation failed: {data.get('error')}")

def cmd_approve(args):
    """Approve plan and start build."""
    result = subprocess.run(
        ["python3", "N5/pulse/review_manager.py", "respond", args.slug, "go"],
        capture_output=True, text=True
    )
    data = json.loads(result.stdout)
    
    if data.get("action") == "approved":
        print(f"✅ Plan approved. Starting build...")
        # Trigger build start
        cmd_start_args = argparse.Namespace(slug=args.slug)
        cmd_start(cmd_start_args)
    else:
        print(f"Response: {data}")
