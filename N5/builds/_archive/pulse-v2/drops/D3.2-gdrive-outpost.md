```python
def sync_build_to_drive(slug: str) -> dict:
    """
    Sync entire build folder to Drive outpost.
    """
    build_dir = Path(f"N5/builds/{slug}")
    
    if not build_dir.exists():
        return {"success": False, "error": "Build not found"}
    
    # Read PLAN.md
    plan_path = build_dir / "PLAN.md"
    if not plan_path.exists():
        return {"success": False, "error": "No PLAN.md found"}
    
    plan_content = plan_path.read_text()
    
    # Upload plan
    upload_result = upload_plan_as_doc(slug, plan_content)
    
    # Update meta.json with sync status
    meta_path = build_dir / "meta.json"
    if meta_path.exists():
        meta = json.loads(meta_path.read_text())
        meta["drive_sync"] = {
            "synced_at": datetime.utcnow().isoformat() + "Z",
            "folder_path": upload_result.get("folder_path"),
            "shareable_link": upload_result.get("shareable_link"),
            "sync_failed": not upload_result.get("success", False)
        }
        meta_path.write_text(json.dumps(meta, indent=2))
    
    return upload_result

def get_shareable_link(slug: str) -> str:
    """Get the shareable link for a build's plan."""
    meta_path = Path(f"N5/builds/{slug}/meta.json")
    
    if not meta_path.exists():
        return None
    
    meta = json.loads(meta_path.read_text())
    return meta.get("drive_sync", {}).get("shareable_link")

def list_synced_builds() -> list:
    """List all builds that have been synced to Drive."""
    builds_dir = Path("N5/builds")
    synced = []
    
    for build_dir in builds_dir.iterdir():
        if build_dir.is_dir():
            meta_path = build_dir / "meta.json"
            if meta_path.exists():
                meta = json.loads(meta_path.read_text())
                if meta.get("drive_sync"):
                    synced.append({
                        "slug": meta.get("slug"),
                        "title": meta.get("title"),
                        "link": meta.get("drive_sync", {}).get("shareable_link"),
                        "synced_at": meta.get("drive_sync", {}).get("synced_at")
                    })
    
    return synced

def main():
    parser = argparse.ArgumentParser(description="Google Drive Outpost Manager")
    subparsers = parser.add_subparsers(dest="command")

    sync_parser = subparsers.add_parser("sync", help="Sync build to Drive")
    sync_parser.add_argument("slug", help="Build slug")

    link_parser.add_argument("slug", help="Build slug")

    args = parser.parse_args()
    
    if args.command == "sync":
        result = sync_build_to_drive(args.slug)
        print(json.dumps(result, indent=2))

        link = get_shareable_link(args.slug)

        print(json.dumps({"builds": builds, "count": len(builds)}, indent=2))
    else:
        parser.print_help()

if __name__ == "__main__":
    main()
```

## Graceful Degradation

If Google Drive is unavailable:
1. Set `drive_sync.sync_failed = true` in meta.json
2. Return error with suggestion to retry
3. Plan generation and build execution should NOT block on Drive sync failure

```python
def sync_with_fallback(slug: str) -> dict:
```
