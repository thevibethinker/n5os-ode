def gather_thread_context(convo_id: str) -> Dict[str, Any]:
    """
    Gather all context for thread close.
    
    NOW INCLUDES:
    - task_info: If this is an action conversation, includes:
      - task_id
      - task_details  
      - assessment (from close_hooks)
      - what_next (from close_hooks)
    """
    context = {
        # ... existing context gathering
    }
    
    # NEW: Check for action conversation
    from N5.task_system.close_hooks import check_action_conversation, assess_task_completion
    
    action_info = check_action_conversation(convo_id)
    if action_info and action_info.get('is_action'):
        task_id = action_info['task_id']
        assessment = assess_task_completion(convo_id, task_id)

            'task_id': task_id,
            'task_details': action_info['task_details'],
            'assessment': assessment,
            'what_next': assessment.get('next_step'),
            'suggested_action': assessment.get('suggested_action')
        }
    
    return context

### 2. Add `complete_action_conversation()` function

```python
def complete_action_conversation(
    convo_id: str,
    task_id: str,
    status: str,  # 'complete', 'partial', 'blocked', 'followup'
    notes: Optional[str] = None,
    create_followup: bool = False
) -> Dict[str, Any]:
    """
    Complete an action conversation and update task system.
    
    Args:
        convo_id: Conversation ID
        task_id: Task ID
        status: User's chosen status
        notes: Optional notes
        create_followup: If True and status='partial', create follow-up task
    
    Returns:
        - success: bool
        - task_updated: bool
        - followup_created: bool
        - followup_task_id: int (if created)
    """
    from N5.task_system.close_hooks import (
        update_task_from_conversation,
        create_followup_task,
        assess_task_completion
    )
    
    result = {'success': False}
    
    # Update the task
    success = update_task_from_conversation(convo_id, task_id, status, notes)
    result['task_updated'] = success
    
    # Create follow-up if requested
    if create_followup and status == 'partial':
        assessment = assess_task_completion(convo_id, task_id)
        if assessment.get('next_step'):
            followup_id = create_followup_task(
                original_task_id=int(task_id),
                next_step=assessment['next_step'],
                estimated_minutes=assessment.get('estimated_minutes')
            )
            result['followup_created'] = True
            result['followup_task_id'] = followup_id
    
    result['success'] = success
    return result
```
