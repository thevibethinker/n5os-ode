## Verification Script

Create and run this verification:

```python
#!/usr/bin/env python3
"""Verify n5os-ode v2 upgrade completeness."""

import os
import subprocess
from pathlib import Path

root = Path('/home/workspace/N5/export/n5os-ode')
errors = []

# 1. Python compilation
for py in root.rglob('*.py'):
    result = subprocess.run(['python3', '-m', 'py_compile', str(py)], 
                          capture_output=True)
    if result.returncode != 0:
        errors.append(f"Compile error: {py}")

# 2. PROJECT_REPO check
result = subprocess.run(['rg', 'PROJECT_REPO', str(root)], capture_output=True)
if result.stdout:
    errors.append(f"PROJECT_REPO still present:\n{result.stdout.decode()}")

# 3. V-specific references
result = subprocess.run(['rg', '-i', 'careerspan|deals\\.db|positions\\.db', 
                        str(root)], capture_output=True)
if result.stdout:
    errors.append(f"V-specific refs found:\n{result.stdout.decode()}")

# 4. LICENSE exists
if not (root / 'LICENSE').exists():
    errors.append("LICENSE file missing")

# 5. Key new files exist
new_files = [
    'N5/scripts/build_lesson_ledger.py',
    'N5/prefs/operations/mece-worker-framework.md',
    'N5/prefs/communication/voice-lessons.md',
    'docs/conversation-end-v5.md'
]
for f in new_files:
    if not (root / f).exists():
        errors.append(f"Missing new file: {f}")

if errors:
    print("VERIFICATION FAILED:")
    for e in errors:
        print(f"  - {e}")
else:
    print("✓ All verification checks passed")
```
