```typescript
case "sendFollowUp": {
  const callId = data.message?.call?.id || "unknown";
  const phone = data.message?.call?.customer?.number || "";
  if (params.confirmed && phone) {
    // Store intent — actual SMS sent in end-of-call-report
    pendingFollowUps.set(callId, { phone, requestedAt: Date.now() });

  }
  break;
}
```
