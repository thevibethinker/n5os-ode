# Drop Brief: Lineage Schema & Folder Scaffolding

## Context

Jettisons need:
1. **A folder structure** — same as regular builds but in `N5/builds/j-<slug>/`
2. **Lineage tracking** — meta.json records where this build came from
3. **Inherited learnings** — copy relevant lessons from parent build

This Drop creates the scaffolding function. D1.1 will call it when creating jettisons.

## Requirements

### 2. Scaffold Function

```python
def scaffold_jettison_build(
    slug: str,
    task: str,
    build_type: str,
    parent_slug: Optional[str] = None,
    parent_convo: Optional[str] = None,
    moment: Optional[str] = None
) -> Path:
    """
    Create folder structure for a jettison build.
    
    Returns: Path to the created build directory
    """
    from datetime import datetime, timezone
    
    build_dir = BUILDS_DIR / slug
    
    # Create directory structure
    build_dir.mkdir(parents=True, exist_ok=True)
    (build_dir / "drops").mkdir(exist_ok=True)
    (build_dir / "deposits").mkdir(exist_ok=True)
    (build_dir / "artifacts").mkdir(exist_ok=True)
    
    # Determine parent type
    parent_type = None
    if parent_slug:
        # Check if parent is itself a jettison
        parent_meta_path = BUILDS_DIR / parent_slug / "meta.json"
        if parent_meta_path.exists():
            with open(parent_meta_path) as f:
                parent_meta = json.load(f)
            parent_type = "jettison" if parent_meta.get("launch_mode") == "jettison" else "build"
        else:
            parent_type = "build"  # Assume build even if not found
    elif parent_convo:
        parent_type = "conversation"
    
    # Build lineage
    lineage = {
        "parent_type": parent_type,
        "parent_ref": parent_slug or parent_convo,
        "parent_conversation": parent_convo,
        "moment": moment or f"Jettisoned: {task}",
        "branched_at": datetime.now(timezone.utc).isoformat()
    }
    
    # Create meta.json
    meta = {
        "slug": slug,
        "title": task[:100],  # Truncate long tasks
        "build_type": build_type,
        "launch_mode": "jettison",
        "status": "pending",
        "total_streams": 1,
        "current_stream": 0,
        "model": "anthropic:claude-sonnet-4-20250514",
        "lineage": lineage,
        "drops": {
            "D1.1": {
                "name": "jettison-task",
                "stream": 1,
                "depends_on": [],
                "spawn_mode": "manual",
                "status": "pending"
            }
        },
        "currents": {},
        "created_at": datetime.now(timezone.utc).isoformat()
    }
    
    with open(build_dir / "meta.json", "w") as f:
        json.dump(meta, f, indent=2)
    
    # Create STATUS.md
    status_content = f"""---
created: {datetime.now().strftime('%Y-%m-%d')}
last_edited: {datetime.now().strftime('%Y-%m-%d')}
version: 1.0
provenance: {parent_convo or 'jettison'}
---

# Jettison: {slug}

**Type:** {build_type}
**Status:** pending
**Parent:** {parent_slug or parent_convo or 'standalone'}

## Task

{task}

## Progress

- [ ] D1.1: Execute jettison task

## Lineage

{f"Branched from: `{parent_slug}`" if parent_slug else f"Branched from conversation: `{parent_convo}`" if parent_convo else "Standalone jettison"}

Moment: {moment or 'N/A'}
"""
    (build_dir / "STATUS.md").write_text(status_content)
    
    # Inherit learnings from parent if applicable
    if parent_slug:
        inherit_learnings(slug, parent_slug)
    
    return build_dir
```

### 3. Learnings Inheritance

```python
def inherit_learnings(slug: str, parent_slug: str):
    """Copy relevant learnings from parent build to jettison."""
    parent_lessons_path = BUILDS_DIR / parent_slug / "BUILD_LESSONS.json"
    jettison_lessons_path = BUILDS_DIR / slug / "BUILD_LESSONS.json"
    
    if not parent_lessons_path.exists():
        # No learnings to inherit
        return
    
    with open(parent_lessons_path) as f:
        parent_lessons = json.load(f)
    
    # Create inherited lessons with attribution
    inherited = {
        "inherited_from": parent_slug,
        "inherited_at": datetime.now(timezone.utc).isoformat(),
        "lessons": parent_lessons.get("lessons", []),
        "note": "These lessons were inherited from the parent build. Add jettison-specific lessons below.",
        "jettison_lessons": []
    }
    
    with open(jettison_lessons_path, "w") as f:
        json.dump(inherited, f, indent=2)
```

### 4. Helper: Get Lineage Chain

For future use by D2.1 (visualization):

```python
def get_lineage_chain(slug: str) -> list[dict]:
    """
    Walk up the lineage chain from a build to its root.
    Returns list of {slug, type, parent} dicts from root to leaf.
    """
    chain = []
    current = slug
    visited = set()  # Prevent cycles
    
    while current and current not in visited:
        visited.add(current)
        meta_path = BUILDS_DIR / current / "meta.json"
        
        if not meta_path.exists():
            break
            
        with open(meta_path) as f:
            meta = json.load(f)
        
        lineage = meta.get("lineage", {})
        chain.append({
            "slug": current,
            "type": meta.get("build_type"),
            "launch_mode": meta.get("launch_mode"),
            "status": meta.get("status"),
            "parent_ref": lineage.get("parent_ref"),
            "parent_type": lineage.get("parent_type")
        })
        
        # Move to parent
        parent_ref = lineage.get("parent_ref")
        if lineage.get("parent_type") in ("build", "jettison"):
            current = parent_ref
        else:
            break  # Conversation or None - end of chain
    
    return list(reversed(chain))  # Root first
```
