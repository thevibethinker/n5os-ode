# Drop Brief: Lineage Visualization Command

**Mission:** Add `pulse lineage` command that visualizes the DAG of build relationships as an ASCII tree.

**Output:** Modified `Skills/pulse/scripts/pulse.py` with `lineage` subcommand

---

## Context

With jettisons creating parent-child relationships between builds, we need a way to visualize this DAG. This is like `git log --graph` but for builds.

D1.2 already implemented `get_lineage_chain()` for walking up the tree. This Drop adds the full visualization.

### 2. Build Lineage Graph

```python
def build_lineage_graph() -> dict:
    """
    Scan all builds and construct parent->children adjacency graph.
    
    Returns: {
        "nodes": {slug: {type, status, launch_mode, parent}},
        "roots": [slugs with no parent],
        "children": {parent_slug: [child_slugs]}
    }
    """
    nodes = {}
    children = {}  # parent -> [children]
    roots = []
    
    # Scan all build directories
    for build_dir in BUILDS_DIR.iterdir():
        if not build_dir.is_dir():
            continue
        
        meta_path = build_dir / "meta.json"
        if not meta_path.exists():
            continue
        
        try:
            with open(meta_path) as f:
                meta = json.load(f)
        except:
            continue
        
        slug = meta.get("slug", build_dir.name)
        lineage = meta.get("lineage", {})
        parent_ref = lineage.get("parent_ref")
        parent_type = lineage.get("parent_type")
        
        nodes[slug] = {
            "type": meta.get("build_type", "unknown"),
            "status": meta.get("status", "unknown"),
            "launch_mode": meta.get("launch_mode", "orchestrated"),
            "title": meta.get("title", "")[:40],
            "parent_ref": parent_ref if parent_type in ("build", "jettison") else None,
            "parent_type": parent_type
        }
        
        # Track parent-child relationship
        if parent_ref and parent_type in ("build", "jettison"):
            if parent_ref not in children:
                children[parent_ref] = []
            children[parent_ref].append(slug)
        else:
            roots.append(slug)
    
    return {
        "nodes": nodes,
        "roots": sorted(roots),
        "children": children
    }
```

### 3. ASCII Tree Rendering

```python
def render_lineage_tree(graph: dict, root_slug: str = None, depth: int = 10) -> str:
    """
    Render lineage as ASCII tree.
    
    If root_slug provided, show only that subtree.
    Otherwise show all trees.
    """
    STATUS_ICONS = {
        "complete": "✓",
        "running": "●",
        "pending": "○",
        "failed": "✗",
        "dead": "☠",
        "blocked": "⊘"
    }
    
    TYPE_COLORS = {
        "code_build": "",
        "research": "[R]",
        "content": "[C]",
        "planning": "[P]",
        "jettison": "[J]"
    }
    
    lines = []
    
    def render_node(slug: str, prefix: str = "", is_last: bool = True, current_depth: int = 0):
        if current_depth > depth:
            return
        
        node = graph["nodes"].get(slug, {})
        status_icon = STATUS_ICONS.get(node.get("status", "unknown"), "?")
        type_tag = TYPE_COLORS.get(node.get("type", ""), "")
        launch_tag = "⚡" if node.get("launch_mode") == "jettison" else ""
        
        # Build the connector
        connector = "└── " if is_last else "├── "
        
        # Format: slug [type] status_icon
        line = f"{prefix}{connector}{slug} {type_tag}{launch_tag} {status_icon}"
        lines.append(line)
        
        # Recurse to children
        child_slugs = graph["children"].get(slug, [])
        child_prefix = prefix + ("    " if is_last else "│   ")
        
        for i, child in enumerate(sorted(child_slugs)):
            is_child_last = (i == len(child_slugs) - 1)
            render_node(child, child_prefix, is_child_last, current_depth + 1)
    
    # Determine what to render
    if root_slug:
        if root_slug in graph["nodes"]:
            # Show ancestry first
            chain = get_lineage_chain(root_slug)
            if len(chain) > 1:
                lines.append("Ancestry:")
                for i, node in enumerate(chain[:-1]):
                    indent = "  " * i
                    lines.append(f"{indent}└── {node['slug']}")
                lines.append("")
            
            # Show the requested node and its descendants
            lines.append(f"Subtree of {root_slug}:")
            render_node(root_slug, "", True, 0)
        else:
            lines.append(f"Build not found: {root_slug}")
    else:
        # Show all trees
        lines.append("Build Lineage")
        lines.append("=" * 40)
        lines.append("")
        
        for i, root in enumerate(graph["roots"]):
            is_last = (i == len(graph["roots"]) - 1)
            render_node(root, "", is_last, 0)
            if not is_last:
                lines.append("")  # Space between trees
        
        lines.append("")
        lines.append("Legend: ✓ complete  ● running  ○ pending  ✗ failed  ⚡ jettison")
        lines.append("Types: [R] research  [C] content  [P] planning")
    
    return "\n".join(lines)
```

### 4. JSON Output

```python
def render_lineage_json(graph: dict, root_slug: str = None) -> str:
    """Render lineage as JSON for machine consumption."""
    if root_slug:
        # Filter to just this subtree
        def get_subtree(slug):
            node = graph["nodes"].get(slug, {})
            children = graph["children"].get(slug, [])
            return {
                "slug": slug,
                **node,
                "children": [get_subtree(c) for c in children]
            }
        
        if root_slug in graph["nodes"]:
            result = get_subtree(root_slug)
        else:
            result = {"error": f"Build not found: {root_slug}"}
    else:
        # Full graph
        result = {
            "roots": [
                {
                    "slug": root,
                    **graph["nodes"].get(root, {}),
                    "children": [graph["nodes"].get(c, {}) | {"slug": c} for c in graph["children"].get(root, [])]
                }
                for root in graph["roots"]
            ],
            "total_builds": len(graph["nodes"]),
            "total_jettisons": sum(1 for n in graph["nodes"].values() if n.get("launch_mode") == "jettison")
        }
    
    return json.dumps(result, indent=2)
```

### 5. Main Command Handler

```python
def cmd_lineage(args):
    """Show build lineage visualization."""
    graph = build_lineage_graph()
    
    if args.format == "json":
        print(render_lineage_json(graph, args.slug))
    else:
        print(render_lineage_tree(graph, args.slug, args.depth))
```
