## Requirements

### 1. Database Schema

Create `Integrations/recall_ai/models.py` with SQLite schema:

```sql
-- Scheduled bots (calendar → recall mapping)
CREATE TABLE IF NOT EXISTS scheduled_bots (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    bot_id TEXT UNIQUE,                    -- Recall bot ID (set after creation)
    calendar_event_id TEXT NOT NULL,       -- Google Calendar event ID
    meeting_url TEXT NOT NULL,             -- Video call URL
    meeting_title TEXT,                    -- Event title (sanitized)
    meeting_start DATETIME NOT NULL,       -- Original meeting start time
    join_at DATETIME NOT NULL,             -- Bot join time (2 min before)
    status TEXT DEFAULT 'pending',         -- pending, scheduled, joining, recording, done, failed, cancelled
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    error TEXT,
    
    UNIQUE(calendar_event_id, meeting_start)  -- Prevent duplicates
);

CREATE INDEX idx_scheduled_bots_status ON scheduled_bots(status);
CREATE INDEX idx_scheduled_bots_join_at ON scheduled_bots(join_at);
CREATE INDEX idx_scheduled_bots_event_id ON scheduled_bots(calendar_event_id);

-- Bot lifecycle events (webhook history)
CREATE TABLE IF NOT EXISTS bot_events (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    event_id TEXT UNIQUE,                  -- Svix event ID for deduplication
    bot_id TEXT NOT NULL,
    event_type TEXT NOT NULL,              -- bot.status_change, bot.done, etc.
    status_code TEXT,                      -- joining_call, in_call_recording, etc.
    sub_code TEXT,                         -- Additional sub-status
    timestamp DATETIME DEFAULT CURRENT_TIMESTAMP,
    payload JSON,
    processed BOOLEAN DEFAULT FALSE,
    error TEXT,
    
    FOREIGN KEY (bot_id) REFERENCES scheduled_bots(bot_id)
);

CREATE INDEX idx_bot_events_bot_id ON bot_events(bot_id);
CREATE INDEX idx_bot_events_type ON bot_events(event_type);
CREATE INDEX idx_bot_events_timestamp ON bot_events(timestamp);

-- Calendar sync state (track last sync)
CREATE TABLE IF NOT EXISTS sync_state (
    id INTEGER PRIMARY KEY CHECK (id = 1),  -- Single row
    last_sync DATETIME,
    last_event_time DATETIME,               -- Most recent event we've seen
    sync_token TEXT,                        -- Google Calendar sync token
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP
);
```

### 2. Python Model Class

```python
# Integrations/recall_ai/models.py

class RecallDatabase:
    def __init__(self, db_path: str = None):
        self.db_path = db_path or DB_PATH
        self._init_db()
    
    # Scheduled Bots
    def create_scheduled_bot(self, calendar_event_id, meeting_url, meeting_title, meeting_start, join_at) -> int
    def update_scheduled_bot(self, id, **kwargs) -> bool
    def get_scheduled_bot_by_event(self, calendar_event_id, meeting_start) -> dict | None
    def get_scheduled_bot_by_bot_id(self, bot_id) -> dict | None
    def get_pending_bots(self, before: datetime = None) -> list[dict]
    def get_bots_by_status(self, status: str) -> list[dict]
    def cancel_bot(self, id) -> bool
    
    # Bot Events
    def log_event(self, event_id, bot_id, event_type, status_code=None, sub_code=None, payload=None) -> int
    def get_events_for_bot(self, bot_id) -> list[dict]
    def mark_event_processed(self, event_id) -> bool
    def event_exists(self, event_id) -> bool
    
    # Sync State
    def get_sync_state(self) -> dict
    def update_sync_state(self, last_sync=None, last_event_time=None, sync_token=None) -> bool
```
