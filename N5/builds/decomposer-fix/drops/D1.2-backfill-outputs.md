### 1. Normalize Hardik's `scores_complete.json`

Write a Python script to transform:

```python
#!/usr/bin/env python3
"""Normalize Hardik scores to canonical schema."""

import json
from pathlib import Path

INPUT = Path("/home/workspace/Careerspan/meta-resumes/inbox/hardik-flowfuse/scores_complete.json")
OUTPUT = INPUT  # Overwrite in place (but backup first)
BACKUP = INPUT.with_suffix('.json.bak')

def normalize_entry(old: dict) -> dict:
    """Transform old schema to canonical."""
    
    # Parse "4/5" -> 4, 5
    req_score = old.get("required_score", "4/5")
    if isinstance(req_score, str) and "/" in req_score:
        parts = req_score.split("/")
        required_score = int(parts[0])
        max_score = int(parts[1])
    else:
        required_score = int(req_score) if req_score else 4
        max_score = 5
    
    # Parse importance "10/10" -> 10, 10
    imp = old.get("importance", "10/10")
    if isinstance(imp, str) and "/" in imp:
        parts = imp.split("/")
        importance = int(parts[0])
        max_importance = int(parts[1])
    else:
        importance = int(imp) if imp else 10
        max_importance = 10
    
    # Build support array from story_id + evidence_rating
    support = []
    if old.get("story_id"):
        support.append({
            "source": old["story_id"],
            "type": old.get("evidence_rating", "Direct"),
            "score": None,
            "rating": old.get("rating", "Good")
        })
    
    return {
        "skill_name": old.get("skill", old.get("skill_name", "UNKNOWN")),
        "category": old.get("category", "Hard Skill"),
        "rating": old.get("rating", "Good"),
        "required_level": old.get("required_level", "Intermediate"),
        "required_score": required_score,
        "max_score": max_score,
        "importance": importance,
        "max_importance": max_importance,
        "evidence_type": old.get("evidence_type", "Profile"),
        "our_take": old.get("our_take", ""),
        "support": support,
        "source_page": old.get("source_page", None)
    }

def main():
    # Backup
    with open(INPUT) as f:
        original = json.load(f)
    with open(BACKUP, 'w') as f:
        json.dump(original, f, indent=2)
    print(f"Backed up to {BACKUP}")
    
    # Transform
    normalized = [normalize_entry(entry) for entry in original]
    
    # Write
    with open(OUTPUT, 'w') as f:
        json.dump(normalized, f, indent=2, ensure_ascii=False)
    
    print(f"Normalized {len(normalized)} entries")
    print(f"Output: {OUTPUT}")

if __name__ == "__main__":
    main()
```

Run this script to normalize Hardik.
