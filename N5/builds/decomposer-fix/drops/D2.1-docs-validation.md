### 3. Create `Skills/careerspan-decomposer/scripts/validate_output.py`

```python
#!/usr/bin/env python3
"""
Validate decomposer output against canonical schema.

Usage:
    python3 validate_output.py --dir /path/to/decomposed/output

Returns exit code 0 if valid, 1 if errors found.
"""

import argparse
import json
import sys
from pathlib import Path

try:
    import jsonschema
except ImportError:
    print("ERROR: jsonschema not installed. Run: pip install jsonschema")
    sys.exit(1)

SCHEMA_PATH = Path(__file__).parent.parent / "assets" / "canonical_schema.json"

def validate_scores(scores_path: Path, schema: dict) -> list[str]:
    """Validate scores_complete.json against schema."""
    errors = []
    
    if not scores_path.exists():
        errors.append(f"Missing: {scores_path}")
        return errors
    
    with open(scores_path) as f:
        try:
            scores = json.load(f)
        except json.JSONDecodeError as e:
            errors.append(f"Invalid JSON: {e}")
            return errors
    
    if not isinstance(scores, list):
        errors.append(f"scores_complete.json must be array, got {type(scores).__name__}")
        return errors
    
    # Validate each entry
    for i, entry in enumerate(scores):
        try:
            jsonschema.validate(entry, schema["items"])
        except jsonschema.ValidationError as e:
            errors.append(f"Entry {i} ({entry.get('skill_name', 'UNKNOWN')}): {e.message}")
    
    return errors

def validate_overview(overview_path: Path) -> list[str]:
    """Check overview.yaml has required fields."""
    errors = []
    
    if not overview_path.exists():
        errors.append(f"Missing: {overview_path}")
        return errors
    
    import yaml
    with open(overview_path) as f:
        overview = yaml.safe_load(f)
    
    # Check careerspan_score
    score = overview.get("careerspan_score")
    if score is None:
        errors.append("overview.yaml: careerspan_score is null (should be extracted)")
    elif isinstance(score, dict):
        if score.get("overall") is None:
            errors.append("overview.yaml: careerspan_score.overall is missing")
    
    return errors

def validate_manifest(manifest_path: Path) -> list[str]:
    """Check manifest.yaml exists and has required fields."""
    errors = []
    
    if not manifest_path.exists():
        errors.append(f"Missing: {manifest_path}")
        return errors
    
    import yaml
    with open(manifest_path) as f:
        manifest = yaml.safe_load(f)
    
    required = ["candidate", "company", "version"]
    for field in required:
        if field not in manifest and manifest.get(field) is None:
            errors.append(f"manifest.yaml: missing {field}")
    
    return errors

def main():
    parser = argparse.ArgumentParser(description="Validate decomposer output")
    parser.add_argument("--dir", required=True, help="Path to decomposed output directory")
    parser.add_argument("--verbose", "-v", action="store_true", help="Show detailed output")
    args = parser.parse_args()
    
    output_dir = Path(args.dir)
    if not output_dir.exists():
        print(f"ERROR: Directory not found: {output_dir}")
        sys.exit(1)
    
    # Load schema
    if not SCHEMA_PATH.exists():
        print(f"ERROR: Schema not found: {SCHEMA_PATH}")
        sys.exit(1)
    
    with open(SCHEMA_PATH) as f:
        schema = json.load(f)
    
    all_errors = []
    
    # Validate scores
    scores_path = output_dir / "scores_complete.json"
    errors = validate_scores(scores_path, schema)
    if errors:
        all_errors.extend([f"[scores] {e}" for e in errors])
    elif args.verbose:
        with open(scores_path) as f:
            count = len(json.load(f))
        print(f"✓ scores_complete.json: {count} entries valid")
    
    # Validate overview
    overview_path = output_dir / "overview.yaml"
    errors = validate_overview(overview_path)
    if errors:
        all_errors.extend([f"[overview] {e}" for e in errors])
    elif args.verbose:
        print(f"✓ overview.yaml: valid")
    
    # Validate manifest
    manifest_path = output_dir / "manifest.yaml"
    errors = validate_manifest(manifest_path)
    if errors:
        all_errors.extend([f"[manifest] {e}" for e in errors])
    elif args.verbose:
        print(f"✓ manifest.yaml: valid")
    
    # Report
    if all_errors:
        print(f"\n❌ VALIDATION FAILED: {len(all_errors)} error(s)\n")
        for error in all_errors:
            print(f"  • {error}")
        sys.exit(1)
    else:
        print(f"✓ Validation passed: {output_dir.name}")
        sys.exit(0)

if __name__ == "__main__":
    main()
```
