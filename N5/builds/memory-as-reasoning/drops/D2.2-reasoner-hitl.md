### 1. Implement N5Reasoner
Create `N5/cognition/reasoner.py`:

```python
import os
import json
import requests
from typing import List, Dict, Optional
from belief_store import N5BeliefStore
from surprisal_detector import SurprisalEvent

class N5Reasoner:
    def __init__(self, belief_store: N5BeliefStore):
        self.beliefs = belief_store
        self.model = "anthropic:claude-sonnet-4-20250514"  # or from feature flags
    
    async def reason_about(self, event: SurprisalEvent, 
                           context: List[Dict]) -> ReasoningResult:
        """
        Given a surprisal event, use abductive reasoning to infer
        what this means about V's current state/preferences.
        """
        # Get relevant beliefs
        relevant_beliefs = self.beliefs.get_beliefs(min_confidence=0.3)
        
        prompt = self._build_prompt(event, context, relevant_beliefs)
        result = await self._call_zo_ask(prompt)
        return self._parse_result(result)
    
    def _build_prompt(self, event: SurprisalEvent, 
                      context: List[Dict],
                      beliefs: List[Dict]) -> str:
        return f"""
You are analyzing a "surprisal event" - a search query that produced unexpected results.

## The Event
- **Query:** {event.query}
- **Top similarity score:** {event.similarity:.3f} (threshold is 0.3)
- **Reason for surprisal:** {event.reason}

## Top Retrieved Results
{self._format_context(context[:3])}

## Current Beliefs About V
{self._format_beliefs(beliefs[:10])}

## Your Task
Using abductive reasoning (inference to the best explanation), determine:

1. **Inference:** What does this mismatch suggest about V's current focus, priorities, or state?
2. **Belief Updates:** Should any existing beliefs have their confidence adjusted?
3. **New Beliefs:** Should any new beliefs be proposed? (Keep confidence conservative, 0.4-0.6)

Respond in this exact JSON format:
```json
{{
  "inference": "Your inference about what this means",
  "belief_updates": [
    {{"belief_id": "xxx", "confidence_delta": 0.1, "reason": "why"}}
  ],
  "new_beliefs": [
    {{"content": "The belief statement", "confidence": 0.5, "domain": "preference", "reason": "why"}}
  ]
}}
```

Domains: identity | preference | behavior | goal

Be conservative with new beliefs - only propose if there's clear signal.
"""
    
    async def _call_zo_ask(self, prompt: str) -> Dict:
        """Call the /zo/ask API for reasoning."""
        response = requests.post(
            "https://api.zo.computer/zo/ask",
            headers={
                "authorization": os.environ["ZO_CLIENT_IDENTITY_TOKEN"],
                "content-type": "application/json"
            },
            json={
                "input": prompt,
                "output_format": {
                    "type": "object",
                    "properties": {
                        "inference": {"type": "string"},
                        "belief_updates": {"type": "array"},
                        "new_beliefs": {"type": "array"}
                    },
                    "required": ["inference", "belief_updates", "new_beliefs"]
                }
            }
        )
        return response.json()["output"]
    
    def apply_reasoning(self, result: ReasoningResult):
        """Apply reasoning outcomes to belief store."""
        for update in result.belief_updates:
            self.beliefs.update_confidence(
                update["belief_id"], 
                update["confidence_delta"]
            )
        
        for new in result.new_beliefs:
            self.beliefs.add_belief(
                content=new["content"],
                confidence=new["confidence"],
                domain=new["domain"],
                source="inferred"
            )
```

from N5.cognition.belief_store import N5BeliefStore
from N5.cognition.surprisal_detector import SurprisalDetector
from N5.cognition.reasoner import N5Reasoner

# In __init__, add:
if self._load_feature_flag("N5_REASONING_ENABLED", False):
    self.belief_store = N5BeliefStore()
    self.surprisal_detector = SurprisalDetector(
        threshold=self._load_feature_flag("N5_SURPRISAL_THRESHOLD", 0.3)
    )
    self.reasoner = N5Reasoner(self.belief_store)
else:
    self.belief_store = None
    self.surprisal_detector = None
    self.reasoner = None

# In search(), after getting results:
if self.surprisal_detector:
    surprisal = self.surprisal_detector.detect(query, results)
    
    # Log the trace regardless
    self.surprisal_detector.log_trace(query, results, surprisal)
    
    # If surprisal detected and reasoning enabled, invoke reasoner
    if surprisal and surprisal.needs_reasoning and self.reasoner:
        try:
            # Run reasoning asynchronously (don't block search return)
            import asyncio
            asyncio.create_task(self._reason_async(surprisal, results))
        except Exception as e:
            LOG.warning(f"Reasoning failed: {e}")

async def _reason_async(self, surprisal, results):
    """Run reasoning in background."""
    try:
        reasoning_result = await self.reasoner.reason_about(surprisal, results)
        self.reasoner.apply_reasoning(reasoning_result)
        
        # Update the trace with reasoning result
        self.surprisal_detector.log_trace(
            surprisal.query, results, surprisal, 
            reasoning_result=reasoning_result.__dict__
        )
    except Exception as e:
        LOG.error(f"Background reasoning failed: {e}")
```

### 3. Create HITL Review CLI
Create `N5/scripts/review_beliefs.py`:

```python
#!/usr/bin/env python3
"""
HITL Review CLI for low-confidence beliefs.

Usage:
  python3 review_beliefs.py list          # Show pending reviews
  python3 review_beliefs.py approve <id>  # Approve a belief
  python3 review_beliefs.py reject <id>   # Reject a belief
  python3 review_beliefs.py export        # Export snapshot
"""

import argparse
import sys
sys.path.insert(0, '/home/workspace')
from N5.cognition.belief_store import N5BeliefStore

def main():
    parser = argparse.ArgumentParser(description="HITL Belief Review")
    parser.add_argument("command", choices=["list", "approve", "reject", "export"])
    parser.add_argument("belief_id", nargs="?")
    parser.add_argument("--notes", help="Reviewer notes")
    args = parser.parse_args()
    
    store = N5BeliefStore()
    
    if args.command == "list":
        pending = store.get_review_queue(status="pending")
        if not pending:
            print("No pending reviews.")
            return
        for item in pending:
            print(f"\n[{item['id']}] Belief: {item['belief_id']}")
            print(f"  Content: {item['content'][:100]}...")
            print(f"  Confidence: {item['confidence']:.2f}")
            print(f"  Reason: {item['reason']}")
    
    elif args.command == "approve":
        if not args.belief_id:
            print("Error: belief_id required")
            return
        store.approve_review(args.belief_id, notes=args.notes)
        print(f"Approved: {args.belief_id}")
    
    elif args.command == "reject":
        if not args.belief_id:
            print("Error: belief_id required")
            return
        store.reject_review(args.belief_id, notes=args.notes)
        print(f"Rejected: {args.belief_id}")
    
    elif args.command == "export":
        path = store.export_snapshot()
        print(f"Exported to: {path}")

if __name__ == "__main__":
    main()
```
