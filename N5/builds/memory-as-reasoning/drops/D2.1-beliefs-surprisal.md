### 1. Create reasoning.db Schema
Create `N5/cognition/reasoning.db` with this schema:

```sql
-- Beliefs table
CREATE TABLE beliefs (
    id TEXT PRIMARY KEY,
    content TEXT NOT NULL,
    confidence REAL DEFAULT 0.5,
    domain TEXT,  -- identity | preference | behavior | goal
    source TEXT,  -- inferred | explicit | resonance
    evidence_json TEXT,  -- JSON array of block_ids
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    validation_count INTEGER DEFAULT 0,
    status TEXT DEFAULT 'active'  -- active | archived | rejected
);

-- Reasoning traces
CREATE TABLE reasoning_traces (
    id TEXT PRIMARY KEY,
    timestamp DATETIME DEFAULT CURRENT_TIMESTAMP,
    query TEXT,
    top_similarity REAL,
    surprisal_triggered BOOLEAN,
    reasoning_result TEXT,
    belief_updates_json TEXT
);

-- HITL review queue
CREATE TABLE review_queue (
    id TEXT PRIMARY KEY,
    belief_id TEXT NOT NULL,
    reason TEXT,  -- low_confidence | contradiction | manual
    status TEXT DEFAULT 'pending',  -- pending | approved | rejected
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    reviewed_at DATETIME,
    reviewer_notes TEXT,
    FOREIGN KEY(belief_id) REFERENCES beliefs(id)
);

-- Indexes
CREATE INDEX idx_beliefs_domain ON beliefs(domain);
CREATE INDEX idx_beliefs_confidence ON beliefs(confidence);
CREATE INDEX idx_beliefs_status ON beliefs(status);
CREATE INDEX idx_review_status ON review_queue(status);
CREATE INDEX idx_traces_timestamp ON reasoning_traces(timestamp);
```

### 3. Implement SurprisalDetector
Create `N5/cognition/surprisal_detector.py`:

```python
class SurprisalDetector:
    def __init__(self, threshold: float = 0.3):
        self.threshold = threshold
    
    def detect(self, query: str, results: List[Dict]) -> Optional[SurprisalEvent]:
        """
        Check if search results indicate surprisal.
        
        Surprisal occurs when:
        - Top result similarity < threshold (0.3)
        - No results found
        - Results don't match expected domain
        """
        if not results:
            return SurprisalEvent(
                query=query,
                reason="no_results",
                similarity=0.0,
                needs_reasoning=True
            )
        
        top_similarity = results[0].get("score", 0)
        if top_similarity < self.threshold:
            return SurprisalEvent(
                query=query,
                reason="low_similarity",
                similarity=top_similarity,
                needs_reasoning=True
            )
        
        return None
    
    def log_trace(self, query: str, results: List[Dict], 
                  surprisal: Optional[SurprisalEvent],
                  reasoning_result: Dict = None):

    similarity: float
    needs_reasoning: bool
```
