## Key Integration Pattern

All integration scripts should follow this pattern:

```python
from db_paths import get_db_connection

def get_person_context(person_id: int) -> dict:
    """Get full context for a person including deals."""
    conn = get_db_connection(readonly=True)
    c = conn.cursor()
    
    # Get person
    c.execute("SELECT * FROM people WHERE id = ?", (person_id,))
    person = dict(c.fetchone())
    
    # Get deals
    c.execute("""
        SELECT d.company, d.deal_type, dr.role
        FROM deals d
        JOIN deal_roles dr ON d.id = dr.deal_id
        WHERE dr.person_id = ?
    """, (person_id,))
    person['deals'] = [dict(r) for r in c.fetchall()]
    
    # Get recent interactions
    c.execute("""
        SELECT type, summary, occurred_at
        FROM interactions
        WHERE person_id = ?
        ORDER BY occurred_at DESC
        LIMIT 5
    """, (person_id,))
    person['recent_interactions'] = [dict(r) for r in c.fetchall()]
    
    conn.close()
    return person
```
