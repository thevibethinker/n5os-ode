```typescript
    scores: { A: 1, B: 1, C: 2, D: 3 }

  // ... 3 more questions
];

function calculateLevel(responses: Response[]): 1 | 2 | 3 {
  const avg = responses.reduce((sum, r) => sum + r.score, 0) / responses.length;
  if (avg < 1.5) return 1;
  if (avg < 2.5) return 2;
  return 3;
}
```
