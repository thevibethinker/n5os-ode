### 1. Add Gmail Query Execution

Replace `scan_emails()` that returns query strings with actual Gmail API calls:

```python
from use_app_gmail import gmail_find_email

def scan_emails(config: dict) -> list:
    """Scan Gmail for emails matching pipeline criteria."""
    results = []
    for query in config['email']['queries']:
        emails = gmail_find_email(query=query['query'], maxResults=20)
        results.extend(emails)
    return dedupe_and_classify(results)
```

```python
def log_activity(action: str, target: dict, result: dict, config: dict):
    """Log pipeline activity to Airtable."""
    use_app_airtable(
        tool_name="airtable_oauth-create-record",
        configured_props={
            "baseId": config['airtable']['base_id'],
            "tableId": config['airtable']['tables']['activity_log'],
            "fields": {
                "Timestamp": datetime.now().isoformat(),
                "Action": action,
                "Target": json.dumps(target),
                "Result": json.dumps(result),
                "Status": "success" if not result.get('error') else "error"
            }
        },
        email="vrijen@mycareerspan.com"
    )
```
