### 3. Build Review (Final Teaching Summary)

Create `N5/pulse/teaching/build_review.py`:

```python
#!/usr/bin/env python3
"""
Final teaching review at build close.
Summarizes all teaching moments and updates glossary.
"""

def generate_build_teaching_summary(slug: str) -> dict:
    """
    Generate summary of all teaching moments from this build.
    
    Returns:
    {
        "total_moments": 3,
        "new_terms_introduced": ["MECE", "decomposition"],
        "unabsorbed_terms": [...],
        "suggested_review": [...],
        "summary_text": "..."
    }
    """
    moments_path = Path(f"N5/builds/{slug}/teaching_moments.jsonl")
    moments = []
    if moments_path.exists():
        with open(moments_path) as f:
            for line in f:
                if line.strip():
                    moments.append(json.loads(line))
    
    # Analyze moments
    unabsorbed = [m for m in moments if not m.get("absorbed")]
    
    # Generate summary
    if len(unabsorbed) > 0:
        summary = f"📚 This build introduced {len(unabsorbed)} term(s) to review:\n"
        for m in unabsorbed[:3]:  # Show top 3
            term = m.get("moment", {}).get("teaching", {}).get("precise_term", "")
            summary += f"• {term}\n"
    else:
        summary = "✓ No new vocabulary from this build."
    
    return {
        "total_moments": len(moments),
        "new_terms_introduced": [...],
        "unabsorbed_terms": unabsorbed,
        "summary_text": summary
    }

def mark_build_teaching_complete(slug: str):
```
