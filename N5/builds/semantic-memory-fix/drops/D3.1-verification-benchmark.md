### 1. Search Performance Benchmark

Run 5 diverse queries and measure latency:

```python
import sys, time
sys.path.insert(0, '/home/workspace')
from N5.cognition.n5_memory_client import N5MemoryClient

queries = [
    "V's beliefs about AI and career coaching",
    "careerspan business model revenue",
    "P35 P36 architectural principles",
    "meeting notes from January 2026",
    "how to debug semantic memory issues",
]

client = N5MemoryClient()
print(f"Provider: {client.provider}")
print(f"ANN index: {client.ann_index is not None}")
print(f"Reranker: {client.reranker is not None}")

for q in queries:
    t0 = time.time()
    results = client.search(q, limit=5)
    t1 = time.time()
    print(f"\n[{t1-t0:.3f}s] {q}")
    for r in results[:3]:
        print(f"  [{r.get('score',0):.3f}] {r.get('path','?')[len('/home/workspace/'):]}")
```

**Pass criteria**: All queries under 2 seconds. Average under 1 second.

### 3. Data Integrity Check

```python
import sqlite3, os, numpy as np

for db_name, db_path in [
    ('vectors_v2.db', '/home/workspace/N5/cognition/vectors_v2.db'),
    ('brain.db', '/home/workspace/N5/cognition/brain.db')
]:
    conn = sqlite3.connect(db_path)
    cur = conn.cursor()
    
    # Orphan checks
    cur.execute('SELECT COUNT(*) FROM blocks WHERE id NOT IN (SELECT block_id FROM vectors)')
    blocks_no_vec = cur.fetchone()[0]
    cur.execute('SELECT COUNT(*) FROM vectors WHERE block_id NOT IN (SELECT id FROM blocks)')
    vec_no_blocks = cur.fetchone()[0]
    cur.execute('SELECT COUNT(*) FROM resources WHERE id NOT IN (SELECT DISTINCT resource_id FROM blocks)')
    res_no_blocks = cur.fetchone()[0]
    
    # Stale check (sample 500)
    cur.execute('SELECT path FROM resources ORDER BY RANDOM() LIMIT 500')
    sample_paths = [r[0] for r in cur.fetchall()]
    stale_count = sum(1 for p in sample_paths if not os.path.exists(p))
    stale_pct = stale_count / len(sample_paths) * 100 if sample_paths else 0
    
    print(f"\n=== {db_name} ===")
    print(f"Blocks without vectors: {blocks_no_vec} {'✅' if blocks_no_vec == 0 else '❌'}")
    print(f"Vectors without blocks: {vec_no_blocks} {'✅' if vec_no_blocks == 0 else '❌'}")
    print(f"Resources without blocks: {res_no_blocks} {'✅' if res_no_blocks == 0 else '❌'}")
    print(f"Stale resources (sample 500): {stale_pct:.1f}% {'✅' if stale_pct < 2 else '❌'}")
    conn.close()
```

**Pass criteria**: All zeros for orphans. Stale < 2%.

### 7. Coverage Check

```python
import sqlite3, os

db = '/home/workspace/N5/cognition/vectors_v2.db'
conn = sqlite3.connect(db)
cur = conn.cursor()
cur.execute('SELECT DISTINCT path FROM resources')
indexed = set(r[0] for r in cur.fetchall())
conn.close()

for d, name in [
    ('/home/workspace/Personal/', 'Personal/'),
    ('/home/workspace/Knowledge/', 'Knowledge/'),
    ('/home/workspace/Documents/', 'Documents/'),
    ('/home/workspace/N5/prefs/', 'N5/prefs/'),
]:
    total = indexed_count = 0
    for root, dirs, files in os.walk(d):
        dirs[:] = [x for x in dirs if x not in ('.git', 'node_modules', '__pycache__', 'Trash', 'Archive')]
        for f in files:
            if f.endswith('.md'):
                total += 1
                if os.path.join(root, f) in indexed:
                    indexed_count += 1
    pct = (indexed_count/total*100) if total > 0 else 0
    status = '✅' if pct > 85 else '⚠️' if pct > 60 else '❌'
    print(f"{name}: {indexed_count}/{total} ({pct:.0f}%) {status}")
```

**Pass criteria**: Documents/ > 85%. Others maintained at >90%.
